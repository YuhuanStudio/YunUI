import { CodeBlock, InlineCode } from "yunui/content";

A syntax-highlighted code block powered by [Shiki](https://shiki.style). The highlighter is loaded on demand the first time a block renders, so it never ships in the initial bundle. Shows an optional `filename` header, line numbers, highlighted lines, a copy button, and — when you pass `onEdit` — an "Edit" action that hands back the code payload. Pass the code as the child string.

## Import

```tsx
import { CodeBlock, InlineCode } from "@yuhuanowo/yunui/content";
```

## Example

<ComponentPreview
  title="CodeBlock"
  code={`import { CodeBlock } from "@yuhuanowo/yunui/content";`}
>
  <div className="w-full max-w-xl text-left">
    <CodeBlock language="tsx" filename="button.tsx" showLineNumbers highlightLines={[2]}>{`export function Button({ label }: { label: string }) {
  return <button className="btn btn-primary">{label}</button>;
}`}</CodeBlock>
  </div>
</ComponentPreview>
