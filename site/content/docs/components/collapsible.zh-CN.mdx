import { Collapsible, CollapsibleTrigger, CollapsibleContent, Button } from "yunui";
import { ChevronsUpDown } from "lucide-react";

基于 Radix Collapsible 构建的带动画显示/隐藏区域。`CollapsibleTrigger` 切换 `CollapsibleContent` 的可见性，内容会滑动展开和收起。非受控场景使用 `defaultOpen`，需要自行控制时则使用 `open` + `onOpenChange`。

## 导入

```tsx
import {
  Collapsible, CollapsibleTrigger, CollapsibleContent,
} from "@yuhuanowo/yunui";
```

## 基础

<ComponentPreview
  title="Toggle a section"
  code={`<Collapsible>
  <CollapsibleTrigger>Toggle details</CollapsibleTrigger>
  <CollapsibleContent>
    YunUI is a versioned design system: build it once, sync it everywhere.
  </CollapsibleContent>
</Collapsible>`}
>
  <Collapsible className="w-full max-w-md">
    <CollapsibleTrigger className="font-medium underline-offset-4 hover:underline">
      Toggle details
    </CollapsibleTrigger>
    <CollapsibleContent>
      <p className="pt-2 text-sm text-muted-foreground">
        YunUI is a versioned design system: build it once, sync it everywhere.
      </p>
    </CollapsibleContent>
  </Collapsible>
</ComponentPreview>

## 默认展开

通过 `defaultOpen`（非受控）让它在首次渲染时即为展开状态。

<ComponentPreview
  title="Default open"
  code={`<Collapsible defaultOpen>
  <CollapsibleTrigger>
    <Button variant="outline" size="sm">
      What's included <ChevronsUpDown className="w-4 h-4" />
    </Button>
  </CollapsibleTrigger>
  <CollapsibleContent>
    Primitives, layout helpers, and themed tokens.
  </CollapsibleContent>
</Collapsible>`}
>
  <Collapsible defaultOpen className="w-full max-w-md space-y-2">
    <CollapsibleTrigger asChild>
      <Button variant="outline" size="sm">
        What's included <ChevronsUpDown className="w-4 h-4" />
      </Button>
    </CollapsibleTrigger>
    <CollapsibleContent>
      <p className="text-sm text-muted-foreground">
        Primitives, layout helpers, and themed tokens.
      </p>
    </CollapsibleContent>
  </Collapsible>
</ComponentPreview>
