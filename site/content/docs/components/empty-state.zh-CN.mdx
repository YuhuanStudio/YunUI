import { EmptyState, Button } from "yunui";
import { Inbox, SearchX } from "lucide-react";

用于空列表、无结果界面和零数据状态的居中「空空如也」占位符。它依次堆叠可选的 `icon`、必填的 `title`、可选的 `description`，以及可选的 `action` 槽位（例如一个主按钮），让用户知道下一步该做什么。

## 导入

```tsx
import { EmptyState } from "@yuhuanowo/yunui";
```

## 基础

<ComponentPreview
  title="空列表"
  code={`import { Inbox } from "lucide-react";

<EmptyState
  icon={<Inbox className="w-8 h-8" />}
  title="暂无消息"

  />
</ComponentPreview>
