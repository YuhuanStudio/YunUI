import { EmptyState, Button } from "yunui";
import { Inbox, SearchX } from "lucide-react";

用於空清單、無結果畫面和零資料狀態的置中「空空如也」佔位符。它依序堆疊可選的 `icon`、必填的 `title`、可選的 `description`，以及可選的 `action` 插槽（例如一個主按鈕），讓使用者知道下一步該做什麼。

## 匯入

```tsx
import { EmptyState } from "@yuhuanowo/yunui";
```

## 基礎

## 帶操作
