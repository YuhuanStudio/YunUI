import { Collapsible, CollapsibleTrigger, CollapsibleContent, Button } from "yunui";
import { ChevronsUpDown } from "lucide-react";

An animated show/hide region built on Radix Collapsible. A `CollapsibleTrigger` toggles the visibility of a `CollapsibleContent`, which slides open and closed. Use `defaultOpen` for the uncontrolled case, or `open` + `onOpenChange` when you want to drive it yourself.

## Import

```tsx
import {
  Collapsible, CollapsibleTrigger, CollapsibleContent,
} from "@yuhuanowo/yunui";
```

## Open by default

Render it expanded on first paint with `defaultOpen` (uncontrolled).

<ComponentPreview
  title="Default open"
  code={`<Collapsible defaultOpen>
  <CollapsibleTrigger>
    <Button variant="outline" size="sm">
      What's included <ChevronsUpDown className="w-4 h-4" />
    </Button>
  </CollapsibleTrigger>
  <CollapsibleContent>
    Primitives, layout helpers, and themed tokens.
  </CollapsibleContent>
</Collapsible>`}
>
  <Collapsible defaultOpen className="w-full max-w-md space-y-2">
    <CollapsibleTrigger asChild>
      <Button variant="outline" size="sm">
        What's included <ChevronsUpDown className="w-4 h-4" />
      </Button>
    </CollapsibleTrigger>
    <CollapsibleContent>
      <p className="text-sm text-muted-foreground">
        Primitives, layout helpers, and themed tokens.
      </p>
    </CollapsibleContent>
  </Collapsible>
</ComponentPreview>
