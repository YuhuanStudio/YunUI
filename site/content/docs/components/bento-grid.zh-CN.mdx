import { BentoGrid, BentoCard } from "yunui";
import { Sparkles, Zap, Shield } from "lucide-react";

一个响应式的便当（bento）风格网格。`BentoGrid` 布局为单列（移动端）/ 三列（桌面端）网格，每个 `BentoCard` 都是一个玻璃质感的方块，带有头部媒体插槽、图标、标题和描述。通过 `className` 使用 Tailwind 类可让方块跨越多列或多行。

## 导入

```tsx
import { BentoGrid, BentoCard } from "@yuhuanowo/yunui";
```

## 跨列方块

在 `BentoCard` 上使用 `className` 可让其跨越多列，并提供 `header` 在文本上方放置一块媒体区域。

<ComponentPreview
  title="精选磁贴"
  code={`<BentoGrid>
  <BentoCard
    className="md:col-span-2"
    header={<div className="h-24 rounded-xl bg-gradient-to-br from-indigo-400/30 to-purple-400/30" />}
    icon={<Sparkles className="w-5 h-5" />}
    title="精选"
    description="This tile spans two columns and has a header visual."
  />
  <BentoCard
    icon={<Zap className="w-5 h-5" />}
    title="紧凑"
    description="A standard single-column tile."
  />
</BentoGrid>`}
>
  <BentoGrid>
    <BentoCard
      className="md:col-span-2"
      header={<div className="h-24 rounded-xl bg-gradient-to-br from-indigo-400/30 to-purple-400/30" />}
      icon={<Sparkles className="w-5 h-5" />}
      title="精选"
      description="This tile spans two columns and has a header visual."
    />
    <BentoCard
      icon={<Zap className="w-5 h-5" />}
      title="紧凑"
      description="A standard single-column tile."
    />
  </BentoGrid>
</ComponentPreview>
