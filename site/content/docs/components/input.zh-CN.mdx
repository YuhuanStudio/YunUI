import { Input } from "yunui";
import { Search, Mail } from "lucide-react";

单行文本输入框。它呈现标准的输入框样式，支持可选的前置 `icon`，并显示内联的 `error` 错误信息（同时会将边框切换为错误样式）。所有原生 `<input>` 属性都会透传。

## Import

```tsx
import { Input } from "@yuhuanowo/yunui";
```
