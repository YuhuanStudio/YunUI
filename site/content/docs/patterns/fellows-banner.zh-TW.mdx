import { FellowsBanner } from "yunui/patterns";

一個精簡、可點擊的推廣橫幅——帶圖示、標題、描述、可選 `features` 清單（以「·」連接成一行）以及行動呼籲的漸層卡片。整張卡片透過注入的路由轉接器作為連結，點擊時會導向 `href`。

## 匯入

```tsx
import { FellowsBanner } from "@yuhuanowo/yunui/patterns";
```

## 基礎用法

<ComponentPreview
  title="FellowsBanner"
  code={`<FellowsBanner
  title="加入 Fellows 計畫"
  description="Selective access for builders and researchers."
  ctaText="Apply"
  features={["Pro-level limits", "Priority support", "Early features"]}
  href="#"
/>`}
>
  <div className="w-full max-w-xl">
    <FellowsBanner
      title="加入 Fellows 計畫"
      description="Selective access for builders and researchers."
      ctaText="Apply"
      features={["Pro-level limits", "Priority support", "Early features"]}
      href="#"
    />
  </div>
</ComponentPreview>

## 無附加功能

`features` 清單是可選的——省略它即可得到一個極簡橫幅。

  title="極簡"

/>`}

      title="升級到 Pro"

      ctaText="Learn more"

  </div>
</ComponentPreview>
