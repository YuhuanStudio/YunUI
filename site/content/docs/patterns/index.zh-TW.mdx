import { FellowsBanner } from "yunui/patterns";

模式是基於基礎元件構建的、由屬性驅動的高層組合——那些反覆出現的頁面級片段（應用框架、儀表板、部落格、AI 產品 UI）。它們位於 `@yuhuanowo/yunui/patterns` 和 `@yuhuanowo/yunui/ai` 之下，並透過[轉接器](/docs)接收其框架相依（路由、圖片、i18n）。

## 匯入

```tsx
import { StatCard, Sidebar, Footer, PageHeader, FAQ } from "@yuhuanowo/yunui/patterns";
import { ModelCard, ThinkingBlock, ProviderIcon } from "@yuhuanowo/yunui/ai";
```
