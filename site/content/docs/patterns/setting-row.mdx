import { SettingRow } from "yunui/patterns";

One labelled row in a **settings / preferences** list: a title, an optional description, and a trailing control slot — a `Switch`, a `Select`, a button. Stack several inside a card; the bottom border divides them and the last one drops it. Presentation only — the host owns the control and its state.

## Import

```tsx
import { SettingRow } from "@yuhuanowo/yunui/patterns";
```

## Example

Pass any control into the `control` slot. The `description` wraps under the title and is optional.

<ComponentPreview
  title="SettingRow"
  code={`import { SettingRow } from "@yuhuanowo/yunui/patterns";
import { Switch } from "@yuhuanowo/yunui";

<div className="rounded-xl border border-border bg-card px-4">
  <SettingRow title="Email notifications"
    description="Receive a summary of account activity in your inbox."
    control={<Switch checked={email} onCheckedChange={setEmail} />} />
  <SettingRow title="Weekly digest"
    control={<Switch checked={digest} onCheckedChange={setDigest} />} />
</div>`}
>

</ComponentPreview>
