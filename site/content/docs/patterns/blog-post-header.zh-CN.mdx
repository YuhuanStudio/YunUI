import { BlogPostHeader } from "yunui/patterns";

单篇博客文章页面的页头区块：分类徽章、大标题、引导描述、作者/日期/阅读时长元信息行、可选封面图，以及标签链接。

## 引入

```tsx
import { BlogPostHeader } from "@yuhuanowo/yunui/patterns";
```
